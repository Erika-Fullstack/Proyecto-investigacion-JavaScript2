Funciones de Orden Superior

map()
        
          const numeros = [1, 2, 3]; 
          const duplicados = numeros.map(num => num * 2);
          console.log(duplicados); // [2, 4, 6]
       

      

      
      
filter()


      
      

        
        
          const numeros = [1, 2, 3, 4];
          const pares = numeros.filter(num => num % 2 === 0); 
          console.log(pares); // [2, 4]
