¿Qué es un objeto?
            
              class Persona {
                constructor(nombre, apellido, edad) {
                  this.nombre = nombre;
                  this.apellido = apellido;
                  this.edad = edad;
                }
              }
              
              const persona = new Persona("Gorka", "Pérez", 30);
              console.log(persona); // Persona { nombre: 'Gorka', apellido: 'Pérez',edad: 30 }
Uso de "This"
          
            class Persona {
              constructor(nombre, edad) {
                this.nombre = nombre; // "this" se refiere a la instancia actual
                this.edad = edad;
              }
            
              presentarse() {
                console.log(`Me llamo ${this.nombre}`);
              }
            }
            
            const persona1 = new Persona("Pedro", 30);
            persona1.presentarse(); // Me llamo Pedro
Uso de Constructoctores
            
              class Persona {
                constructor(nombre, edad) {
                  this.nombre = nombre; // Inicializa la propiedad nombre
                  this.edad = edad;     // Inicializa la propiedad edad
                }
              }
              
              const persona1 = new Persona("Pedro", 30);
              console.log(persona1.nombre); // Pedro
              console.log(persona1.edad);   // 30
Constructor con herencia
            
              class Animal {
                constructor(nombre) {
                  this.nombre = nombre; // Propiedad inicializada por el constructor
                }
              }
              
              class Perro extends Animal {
                constructor(nombre, raza) {
                  super(nombre); // Llama al constructor de la clase base
                  this.raza = raza; // Propiedad adicional
                }
              }
              
              const miPerro = new Perro("Thor", "Labrador");
              console.log(miPerro.nombre); // Thor
              console.log(miPerro.raza);   // Labrador
