Constructor
            
              class Persona {
                constructor(nombre, edad) {
                  this.nombre = nombre; // Inicializa la propiedad nombre
                  this.edad = edad;     // Inicializa la propiedad edad}
              }
              const persona1 = new Persona("Andrés", 25);
              console.log(persona1.nombre); // Andrés
              console.log(persona1.edad);   // 25
Constructor con herencia
            
              class Animal {
                constructor(nombre) {
                  this.nombre = nombre; // Propiedad inicializada por el constructor}
              }
              
              class Perro extends Animal {
                constructor(nombre, raza) {
                  super(nombre); // Llama al constructor de la clase base
                  this.raza = raza; // Propiedad adicional}
              }
              
              const miPerro = new Perro("Thor", "Labrador");
              console.log(miPerro.nombre); // Thor
              console.log(miPerro.raza);   // Labrador
