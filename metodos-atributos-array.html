Métodos y Atributos de Arrays

.length
        
                      let numeros = [1, 2, 3]; 
                      console.log(numeros.length); // 3

Nos devuelve el número de elementos del array.

.push()
        
                      let numeros = [1, 2, 3]; 
                      numeros.push(4); // [1, 2, 3, 4]

Agrega un elemento al final.

.pop()
        
                      let numeros = [1, 2, 3, 4]; 
                      numeros.pop(); 
                      // [1, 2, 3]

Elimina un elemento al final. Nos devuelve el elemento eliminado

.unshift()
          
                        let numeros = [1, 2, 3, 4]; 
                        numeros.unshift(0); 
                        // [0, 1, 2, 3, 4];

Agrega un elemento al principio

.shift()
        
                      let numeros = [1, 2, 3] 
                      numeros.shift(); // [2, 3]

Elimina un elemento al principio. Nos devuelve el elemento eliminado

.sort()
        
                      let letras = ["b", "a", "c"]; 
                      console.log(letras.sort()); // ["a", "b", "c"]
.reverse()
        
                      let numeros = [1, 2, 3]; 
                      console.log(numeros.reverse()); 
                      // [3, 2, 1]
.slice()
        
                      let seleccion = frutas.slice(0, 2);
                      // ["manzana", "banana"]

                    let frutas2 = ["banana", "uva"];
                    let todas = frutas.concat(frutas2);
                    // ["manzana", "banana", "banana", "uva"]

                    frutas = ["manzana", "banana", "banana", "uva"]
