Métodos y Atributos de Arrays

.length
        
          let numeros = [1, 2, 3]; 
          console.log(numeros.length); // 3
.push()
        
          let numeros = [1, 2]; 
          numeros.push(3); // [1, 2, 3]
.unshift()
          
            let numeros = [2, 3]; 
            numeros.unshift(1); // [1, 2, 3]
.shift()
        
          let numeros = [1, 2, 3] 
          numeros.shift(); // [2, 3]
.sort()
        
          let letras = ["b", "a", "c"]; 
          console.log(letras.sort()); // ["a", "b", "c"]
.reverse()
        
          let numeros = [1, 2, 3]; 
          console.log(numeros.reverse()); // [3, 2, 1]
.map()
        
          let numeros = [1, 2, 3]; 
          console.log(numeros.map(x => x * 2)); // [2, 4, 6]
.filter()
        
          let numeros = [1, 2, 3, 4]; 
          console.log(numeros.filter(x => x % 2 === 0)); // [2, 4]
.reduce()
        
          let numeros = [1, 2, 3]; 
          console.log(numeros.reduce((acc, x) => acc + x, 0)); // 6
.includes()
        
          let numeros = [1, 2, 3]; 
          console.log(numeros.includes(2)); // true
