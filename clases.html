Clase Date

            let christmas = new Date('2024-12-25T00:00:00');

Métodos

Obtener Componentes de Fecha
                
        let date = new Date();

    console.log(date.getFullYear()); // Año actual, ej. 2024
    console.log(date.getMonth());    // Mes (0-11), ej. 11 tipos diciembre
    console.log(date.getDate());     // Día del mes (1-31)
    console.log(date.getDay());      // Día de la semana (0-6), ej. 0 tipos domingo
Obtener Componentes de Hora
                
    console.log(date.getHours());    // Hora (0-23)
    console.log(date.getMinutes());  // Minutos (0-59)
    console.log(date.getSeconds());  // Segundos (0-59)
    console.log(date.getMilliseconds()); // Milisegundos (0-999)
Modificar Componentes de Fecha
                                  
            date.setFullYear(2025);
            date.setMonth(0);    // Enero
            date.setDate(15);    // Día 15
            console.log(date);   // Muestra la nueva fecha ajustada
Comparar Fechas
                                  
    let date1 = new Date('2024-12-25');
    let date2 = new Date('2024-01-01');
                    
    console.log(date1 > date2); 
    // true (25 de diciembre es después del 1 de enero)

    console.log(date1 - date2); 
    // Diferencia en milisegundos

Atributos

Obtener el timestamp Actual
                                  
    let timestamp = Date.now();
    console.log(timestamp); 
    // Milisegundos desde 1970, ej. 1702438452093

Ejemplos con Fechas

Ejemplos
                                  
    let now = new Date();
    console.log("Fecha actual:", now.toString()); // Muestra la fecha actual
                    
    // Cambiar fecha
    now.setFullYear(2025);
    now.setMonth(0); // Enero
    now.setDate(15);
    console.log("Fecha modificada:", now.toDateString());
    // Muestra fecha modificada
                   
    // Compara fechas
    let anotherDate = new Date('2025-01-15');
    console.log("¿Son iguales?", now.getTime() === anotherDate.getTime());
    // true
                  
    // Calcular días entre dos fechas
    let startDate = new Date('2024-12-01');
    let endDate = new Date('2024-12-25');
    let daysDifference = (endDate - startDate) / (1000 * 60 * 60 * 24);
    console.log(`Diferencia en días: ${daysDifference}`); // 24 días

Clase Math

Inastanciación
                                  
        // No se puede hacer esto:
        let mathInstance = new Math(); // Error: Math no es un constructor
                    
        // Uso correcto:
        let piValue = Math.PI; 
        console.log(piValue); // 3.141592653589793

Métodos Principales

Redondeo de Números
                                  
    console.log(Math.round(4.6));   // 5 (redondeo al entero más cercano)
    console.log(Math.ceil(4.2));    // 5 (redondeo hacia arriba)
    console.log(Math.floor(4.8));   // 4 (redondeo hacia abajo)
Potencias y Raíces
                                  
    console.log(Math.pow(2, 3));    // 8 (2 elevado a la 3)
    console.log(Math.sqrt(16));     // 4 (raíz cuadrada de 16)
    console.log(Math.cbrt(27));     // 3 (raíz cúbica de 27)
Generación de Números Aleatorios
                                  
     console.log(Math.random()); // Número aleatorio entre 0 y 1

     // Número aleatorio entre 1 y 10
     let random = Math.floor(Math.random() * 10) + 1;
     console.log(random); // Ejemplo: 7
Valores Absolutos Mínimos y Máximos
                                  
    console.log(Math.abs(-5));        // 5 (valor absoluto)
    console.log(Math.max(1, 5, 10));  // 10 (mayor valor)
    console.log(Math.min(1, 5, 10));  // 1 (menor valor)
Trigonometría
                                  
     console.log(Math.sin(Math.PI / 2)); // 1 (seno de π/2)
     console.log(Math.cos(0));          // 1 (coseno de 0)
    console.log(Math.tan(Math.PI / 4)); // 1 (tangente de π/4)

Array

Instanciación

Constructor Array
                                  
let arr2 = new Array(5);

console.log(arr2); // [ <5 empty items> ]
                
let arr3 = new Array(1, 2, 3); // Crea un array con valores iniciales
console.log(arr3); // [1, 2, 3]
Convertir Estructuras en Arrays
                                  
let str = "Hola";
let arrFromStr = Array.from(str);
// Convierte una cadena en un array de caracteres

console.log(arrFromStr); // ['H', 'o', 'l', 'a']
                    
let arrFromSet = Array.from(new Set([1, 2, 2, 3]));
// Convierte un Set en un array

console.log(arrFromSet); // [1, 2, 3]
